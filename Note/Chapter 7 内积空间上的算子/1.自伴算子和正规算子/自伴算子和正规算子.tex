\tbf{1.伴随}
\begin{definition}[1.1 定义:伴随]
    设$T\in\L(V,W)$,$T$的\tbf{伴随}是使得对任意$v\in V$和任意$w\in W$都有$\langle Tv,w\rangle=\langle v,T^*w\rangle$的映射$T^*:W\to V$.
\end{definition}
